Usage

Quick Start

Get started in seconds with your preferred framework.

HTML 
<!-- Include GlassFlow -->
<link rel="stylesheet" href="glassflow.css"/>
<script src="glassflow.js"></script>
<script src="glassflow-components.js"></script>

<!-- Use glass components -->
<glass-card hoverable>
  <h3>Hello Glass!</h3>
  <p>Liquid glass design.</p>
</glass-card>

<glass-button variant="filled">
  Click Me
</glass-button>

<glass-input label="Email"
  placeholder="you@example.com"/>
React 
import { GlassFlow } from 'glassflow-ui';

// Initialize in your app
useEffect(() => {
  GlassFlow.init({ theme: 'dark' });
}, []);

function App() {
  return (
    <div>
      <glass-card hoverable>
        <h3>React + GlassFlow</h3>
        <p>Seamless integration.</p>
      </glass-card>

      <glass-button
        variant="filled"
        onClick={() => alert('Hi!')}
      >
        Get Started
      </glass-button>
    </div>
  );
}
Vue 
<template>
  <glass-card hoverable>
    <h3>Vue + GlassFlow</h3>
    <p>Works with Vue 3.</p>
  </glass-card>

  <glass-input
    v-model="email"
    label="Email"
    placeholder="you@example.com"
  />

<script setup>
import { onMounted, ref } from 'vue'
import { GlassFlow } from 'glassflow-ui'

onMounted(() => {
  GlassFlow.init({ theme: 'ocean' })
})

const email = ref('')
</script>
JavaScript 
// Initialize with options
const gf = new GlassFlow({
  theme: 'auto',
  blur: 20,
  mouseLightEnabled: true,
}).init();

// Switch theme at runtime
gf.setTheme('aurora');

// Listen for events
gf.on('themechange', (e) => {
  console.log(`Theme: ${e.theme}`);
});

// Programmatic ripple
gf.ripple('.my-element', x, y);

// Spring animation
gf.spring('.my-element', 'transform', 1.05);

// Cleanup
gf.destroy();
